@dp.message(Command("start")) async def cmd_start_with_data(message: types.Message, command: CommandObject): """Обрабатывает /start с данными""" args = command.args if args: try: # Декодируем данные import urllib.parse decoded_data = urllib.parse.unquote(args) data = json.loads(decoded_data) logger.info("=" * 60) logger.info("🔥 ПОЛУЧЕНЫ ДАННЫЕ ЧЕРЕЗ START!") logger.info(f"📦 Данные: {json.dumps(data, indent=2)}") logger.info("=" * 60) await message.answer( f"✅ Данные получены через start!\n" f"User ID: {data.get('user', {}).get('id')}\n" f"start_param: {data.get('telegram', {}).get('start_param')}" ) except Exception as e: logger.error(f"❌ Ошибка парсинга: {e}") await message.answer(f"❌ Ошибка: {e}")
