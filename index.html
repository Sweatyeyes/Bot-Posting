function sendStats(action, details = {}) { if (!window.Telegram?.WebApp) { debugLog('❌ Telegram.WebApp не доступен, статистика не отправлена', 'warn'); return; } try { const statsData = { action: action, user_id: state.userId, post_id: state.postId, is_vip: state.isVip, timestamp: Date.now(), ...details }; debugLog(`📊 Отправка статистики: ${JSON.stringify(statsData)}`); window.Telegram.WebApp.sendData(JSON.stringify(statsData)); } catch (e) { debugLog(`❌ Ошибка отправки статистики: ${e.message}`, 'error'); } }
